/***
 * ───────── doPost: receives expense data or actions from the PWA / Shortcut ─────────
 * 
 * Expected JSON body for logging:
 *   { amount, category, source, paymentMode, tags, comments }
 *
 * Special actions:
 *   { action: "undo" }  →  removes the last logged expense
 ***/
function doPost(e) {
    var lock = LockService.getScriptLock();
    lock.tryLock(10000);
    try {
        Logger.log("doPost: Received request. Payload: " + (e && e.postData ? e.postData.contents : "none"));
        var ss = SpreadsheetApp.getActiveSpreadsheet();
        var p = JSON.parse(e.postData.contents);

        if(p.action === "undo") {
            Logger.log("doPost: Action is undo. Invoking undoLast()");
            return undoLast(ss);
        }

        var sheet = ss.getSheetByName("Expenses");
        var now = new Date();
        var month = Utilities.formatDate(now, TIMEZONE, "yyyy-MM");

        var amount = Number(p.amount) || 0;
        var category = p.category || "Others";
        var source = p.source || "";
        var paymentMode = p.paymentMode || "";
        var tags = p.tags || "";
        var comments = p.comments || "";

        Logger.log("doPost: Appending expense: amount=" + amount + ", category=" + category + ", source=" + source + ", paymentMode=" + paymentMode);
        // Expenses row: Timestamp (Date) | Amount | Category | Source | Payment Mode | Tags | Comments
        sheet.appendRow([now, amount, category, source, paymentMode, tags, comments]);
        SpreadsheetApp.flush(); // Commit the row before reading it back in checkBudgets

        var msg = checkBudgets(ss, month, category);
        return json({
            status: "ok",
            message: msg
        });

    } catch (err) {
        Logger.log("doPost Error: " + err);
        return json({
            status: "error",
            message: "Error: " + err
        });
    } finally {
        lock.releaseLock();
    }
}

/*** Generic safe wrapper — calls fn(), returns a JSON error if it throws.
     Without this, any thrown exception causes Apps Script to return an HTML
     error page (not JSON), which breaks the frontend's res.json() call. ***/
function safeGet(fn, arg) {
    try {
        if (arg !== undefined) {
            return json(fn(arg));
        }
        return json(fn());
    } catch(err) {
        Logger.log("safeGet error [" + fn.name + "]: " + err);
        return json({ status: "error", message: String(err) });
    }
}

/***
 * ───────── doGet: REST-like API, routed by ?action= parameter ─────────
 *
 * Actions:
 *   version   → { version }                         — for PWA version sync
 *   config    → { categories, mapping, paymentModes } — dropdown data
 *   dashboard → { month, totalSpent, ... }           — current month summary
 *   recent    → [ last 20 expenses ]                 — for dashboard list
 *   analytics → { breakdowns + recentExpenses }      — for analytics page
 *   filters   → { categories, sources, tags }        — distinct filter values
 *   search    → [ filtered expenses, max 50 ]        — server-side search
 *   (none)    → health check
 ***/
function doGet(e) {
    var action = (e && e.parameter && e.parameter.action) || "";
    Logger.log("doGet: Received request with action='" + action + "'");

    if(action === "version") {
        return json({
            version: APPSCRIPT_VERSION
        });
    }

    if(action === "config")    return safeGet(getConfigData);
    if(action === "dashboard") return safeGet(getDashboardData);
    if(action === "recent")    return safeGet(getRecentExpenses);
    if(action === "analytics") return safeGet(getAnalyticsData);
    if(action === "filters")   return safeGet(getFilterValues);
    if(action === "search")    return safeGet(searchExpenses, e.parameter);

    // Default health check
    return json({
        status: "ok",
        message: "RupiBook API Running",
        version: APPSCRIPT_VERSION
    });
}

/***
 * ───────── BUDGET CHECK ─────────
 * Called after every expense log. Compares this month's spending against:
 *   - Per-category limit from the Budgets sheet
 *   - Overall monthly limit from Settings!B1
 * Returns a human-readable alert string (shown as a notification on the phone).
 * Also triggers email alerts when budgets are exceeded (with duplicate prevention).
 ***/
function checkBudgets(ss, month, category) {
    Logger.log("checkBudgets: Evaluating budgets for month=" + month + ", category=" + category);
    var settings = ss.getSheetByName("Settings");

    // Default 80% threshold if Settings sheet is missing
    var threshold = 0.8;
    var totalLimit = 0;

    if(settings) {
        var settingsVals = settings.getRange("B1:B2").getValues();
        totalLimit = Number(settingsVals[0][0]) || 0;
        var t = Number(settingsVals[1][0]);
        if(t > 0) {
            threshold = t / 100;
        }
    }

    // Sum all expenses for this month
    var data = ss.getSheetByName("Expenses").getDataRange().getValues();
    var totalSpent = 0;
    var catSpent = 0;

    for(var i = 1; i < data.length; i++) {
        if(!data[i][0] || data[i][1] === "") {
            continue;
        }

        var rowMonth = data[i][7]; // Column H = Month

        if(rowMonth instanceof Date) {
            rowMonth = Utilities.formatDate(rowMonth, TIMEZONE, "yyyy-MM");
        }

        rowMonth = String(rowMonth || "").trim();

        if(rowMonth !== month) {
            continue;
        }

        var amt = Number(data[i][1]) || 0; // Column B = Amount
        totalSpent += amt;

        if(String(data[i][2] || "").trim() === category) { // Column C = Category
            catSpent += amt;
        }
    }

    // Look up this category's limit from the Budgets sheet
    var catLimit = 0;
    var budgets = ss.getSheetByName("Budgets");

    if(budgets) {
        var b = budgets.getDataRange().getValues();
        for(var j = 1; j < b.length; j++) {
            if(String(b[j][0] || "").trim() === String(category || "").trim()) {
                catLimit = Number(b[j][1]) || 0;
                break;
            }
        }
    }

    Logger.log("checkBudgets: totalLimit=" + totalLimit + ", totalSpent=" + totalSpent + ", catLimit=" + catLimit + ", catSpent=" + catSpent + ", threshold=" + (threshold * 100) + "%");

    // Build alert messages
    var alerts = [];

    if(catLimit > 0 && catSpent >= threshold * catLimit) {
        alerts.push(`⚠️ ${category}: ${CURRENCY}${fmt(catSpent)} of ${CURRENCY}${fmt(catLimit)} (${Math.round((catSpent / catLimit) * 100)}%). ${CURRENCY}${fmt(catLimit - catSpent)} left.`);
    }

    if(totalLimit > 0 && totalSpent >= threshold * totalLimit) {
        alerts.push(`🚨 Overall: ${CURRENCY}${fmt(totalSpent)} of ${CURRENCY}${fmt(totalLimit)} (${Math.round((totalSpent / totalLimit) * 100)}%). ${CURRENCY}${fmt(totalLimit - totalSpent)} left.`);
    }

    // ── Email alerts when budgets EXCEED 100% ──
    // Category budget exceeded
    if(catLimit > 0 && catSpent > catLimit) {
        trySendCategoryAlert_(ss, month, category);
    }
    // Overall budget exceeded
    if(totalLimit > 0 && totalSpent > totalLimit) {
        trySendOverallAlert_(ss, month);
    }

    if(alerts.length === 0) {
        return `✅ Logged · ${category}. Month so far: ${CURRENCY}${fmt(totalSpent)}${totalLimit > 0 ? ` / ${CURRENCY}${fmt(totalLimit)}` : ""}.`;
    }

    return alerts.join("\n");
}

/***
 * ───────── Alert deduplication helpers ─────────
 * Uses PropertiesService to track which alerts have been sent.
 * Category alerts: sent once on exceed, then re-sent every 10 days with current status.
 * Overall alerts: sent once on exceed, then re-sent every 10 days.
 ***/
function trySendCategoryAlert_(ss, month, category) {
    var props = PropertiesService.getScriptProperties();
    var key = "alert_cat_" + category + "_" + month;
    var last = props.getProperty(key);
    var now = new Date().getTime();

    if(last) {
        var daysSince = (now - Number(last)) / (1000 * 60 * 60 * 24);
        Logger.log("trySendCategoryAlert_: Checked alert for category '" + category + "'. Days since last alert: " + daysSince.toFixed(1));
        if(daysSince < 10) {
            Logger.log("trySendCategoryAlert_: Skipping alert for category '" + category + "' (already sent within 10 days)");
            return;
        }
    }

    try {
        Logger.log("trySendCategoryAlert_: Sending budget alert email for category: " + category);
        sendCategoryBudgetAlert(ss, month, category);
        props.setProperty(key, String(now));
    } catch(e) {
        Logger.log("Category alert email failed: " + e);
    }
}

function trySendOverallAlert_(ss, month) {
    var props = PropertiesService.getScriptProperties();
    var key = "alert_overall_" + month;
    var last = props.getProperty(key);
    var now = new Date().getTime();

    if(last) {
        var daysSince = (now - Number(last)) / (1000 * 60 * 60 * 24);
        Logger.log("trySendOverallAlert_: Checked overall alert. Days since last alert: " + daysSince.toFixed(1));
        if(daysSince < 10) {
            Logger.log("trySendOverallAlert_: Skipping overall alert (already sent within 10 days)");
            return;
        }
    }

    try {
        Logger.log("trySendOverallAlert_: Sending overall budget alert email");
        sendOverallBudgetAlert(ss, month);
        props.setProperty(key, String(now));
    } catch(e) {
        Logger.log("Overall alert email failed: " + e);
    }
}

/*** ───────── UNDO LAST ENTRY ───────── ***/
function undoLast(ss) {
    Logger.log("undoLast: Initiated last row removal request");
    var sheet = ss.getSheetByName("Expenses");
    var last = sheet.getLastRow();
    if(last > 1) {
        var row = sheet.getRange(last, 1, 1, 8).getValues()[0];
        sheet.deleteRow(last);
        Logger.log("undoLast: Successfully removed row amount=" + row[1] + ", category=" + row[2] + ", timestamp=" + row[0]);
        return json({
            status: "ok",
            message: "↩️ Removed: " + CURRENCY + fmt(row[1]) + " · " + row[2]
        });
    }
    Logger.log("undoLast: No expense row found to delete (sheet is empty or only has headers)");
    return json({
        status: "ok",
        message: "Nothing to undo."
    });
}

/***
 * ───────── AUTOPAY ─────────
 * Set up a daily time-driven trigger for this function:
 *   Triggers → + Add Trigger → logAutopay → Time-driven → Day timer
 *
 * It checks every row in the Autopay sheet. If Active=true and
 * Day of Month matches today, it appends a row to Expenses.
 ***/
function logAutopay() {
    Logger.log("logAutopay: Running autopay log routine...");
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var ap = ss.getSheetByName("Autopay");
    var exp = ss.getSheetByName("Expenses");
    if(!ap || !exp) {
        Logger.log("logAutopay: Required sheets not found");
        return;
    }
    var today = new Date();
    var dayNum = Number(Utilities.formatDate(today, TIMEZONE, "d"));
    var todayStr = Utilities.formatDate(today, TIMEZONE, "yyyy-MM-dd");

    // Pre-read Expenses once (batch) to check for already-logged autopay items today
    var expRows = exp.getDataRange().getValues();
    var rows = ap.getDataRange().getValues();
    var autoLoggedCount = 0;
    for(var i = 1; i < rows.length; i++) {
        var name = String(rows[i][0] || "").trim();
        var amount = rows[i][1];
        var category = rows[i][2];
        var day = Number(rows[i][3]);
        var active = rows[i][4];

        if(active !== true || day !== dayNum || !name) continue;

        // Deduplication: skip if already logged today for this autopay name
        var marker = "Auto-logged: " + name;
        var alreadyLogged = expRows.some(function(r) {
            var ts = r[0] instanceof Date
                ? Utilities.formatDate(r[0], TIMEZONE, "yyyy-MM-dd")
                : String(r[0] || "").slice(0, 10);
            return ts === todayStr && String(r[6] || "") === marker;
        });

        if(!alreadyLogged) {
            Logger.log("logAutopay: Auto-logging active expense: '" + name + "', amount=" + amount + ", category=" + category + ", day=" + day);
            // Append with Date object directly, and omit Month since ARRAYFORMULA handles column H
            exp.appendRow([today, amount, category, name, "Autopay", "autopay", marker]);
            autoLoggedCount++;
        } else {
            Logger.log("logAutopay: Autopay skipped (already logged today): " + name);
        }
    }
    Logger.log("logAutopay: Finished autopay routine. Auto-logged " + autoLoggedCount + " items.");
}

/***
 * ───────── EMAIL SYSTEM ─────────
 *
 * Three types of emails, each with distinct subject patterns for Gmail filtering:
 *   1. [RupiBook] 📊 Monthly Summary — Jun 2026       → 1st of each month (prev month recap)
 *   2. [RupiBook] 🚨 Budget Exceeded — Food — Jun 2026 → when a category exceeds its budget
 *   3. [RupiBook] 🔴 Overall Budget Exceeded — Jun 2026 → when total exceeds monthly limit
 *
 * Trigger setup: create a DAILY time-driven trigger for `dailyTrigger`.
 * Budget alerts (2 & 3) are also fired from checkBudgets() on each expense,
 * with dedup via PropertiesService (re-sent every 10 days max).
 ***/

/*** ── Daily trigger orchestrator ──
 *  Fires monthly summary on the 1st of each month (for the previous month).
 *  Kept as the single function to attach a daily trigger to.
 ***/
function dailyTrigger() {
    var today = new Date();
    var dayOfMonth = Number(Utilities.formatDate(today, TIMEZONE, "d"));
    Logger.log("dailyTrigger: Executing daily trigger check for day of month = " + dayOfMonth);

    // On the 1st, send last month's summary
    if(dayOfMonth === 1) {
        var lastMonth = new Date(today.getFullYear(), today.getMonth() - 1, 1);
        var month = Utilities.formatDate(lastMonth, TIMEZONE, "yyyy-MM");
        Logger.log("dailyTrigger: Day is 1st of month. Triggering monthly summary email for: " + month);
        var ss = SpreadsheetApp.getActiveSpreadsheet();
        try {
            sendMonthlySummary(ss, month);
        } catch(e) {
            Logger.log("Monthly summary email failed: " + e);
        }
    }
}

/***
 * ── Helper: get month expenses with full details ──
 * Returns { total, byCat, bySource, byPayment, expenses[], catLimits }
 ***/
function getMonthDetails_(ss, month) {
    Logger.log("getMonthDetails_: Fetching full details for month=" + month);
    var data = ss.getSheetByName("Expenses").getDataRange().getValues();
    var total = 0;
    var byCat = {};      // category → amount
    var bySource = {};   // source → { amount, count }
    var byPayment = {};  // paymentMode → count
    var expenses = [];   // full expense objects
    var txCount = 0;

    for(var i = 1; i < data.length; i++) {
        if(!data[i][0] || data[i][1] === "") continue;

        var rowMonth = data[i][7];
        if(rowMonth instanceof Date) {
            rowMonth = Utilities.formatDate(rowMonth, TIMEZONE, "yyyy-MM");
        }
        rowMonth = String(rowMonth || "").trim();
        if(rowMonth !== month) continue;

        var amt = Number(data[i][1]) || 0;
        var cat = String(data[i][2] || "Others").trim();
        var src = String(data[i][3] || "").trim();
        var pm = String(data[i][4] || "").trim();

        total += amt;
        txCount++;
        byCat[cat] = (byCat[cat] || 0) + amt;

        if(src) {
            if(!bySource[src]) bySource[src] = { amount: 0, count: 0 };
            bySource[src].amount += amt;
            bySource[src].count++;
        }

        if(pm) {
            byPayment[pm] = (byPayment[pm] || 0) + 1;
        }

        expenses.push({
            timestamp: data[i][0],
            amount: amt,
            category: cat,
            source: src,
            paymentMode: pm,
            tags: String(data[i][5] || "").trim(),
            comments: String(data[i][6] || "").trim()
        });
    }

    // Get category budgets
    var catLimits = {};
    var budgets = ss.getSheetByName("Budgets");
    if(budgets) {
        var b = budgets.getDataRange().getValues();
        for(var j = 1; j < b.length; j++) {
            var cn = String(b[j][0] || "").trim();
            if(cn) catLimits[cn] = Number(b[j][1]) || 0;
        }
    }

    Logger.log("getMonthDetails_: Completed month details fetch. Found " + txCount + " txns, total spent=" + total);
    return {
        total: total,
        txCount: txCount,
        byCat: byCat,
        bySource: bySource,
        byPayment: byPayment,
        expenses: expenses,
        catLimits: catLimits
    };
}

/*** ── Helper: format month label (yyyy-MM → "Jun 2026") ── ***/
function fmtMonthLabel_(month) {
    var parts = month.split("-");
    var names = ["Jan","Feb","Mar","Apr","May","Jun","Jul","Aug","Sep","Oct","Nov","Dec"];
    return names[Number(parts[1]) - 1] + " " + parts[0];
}

/*** ── Helper: category icon ── ***/
function catIcon_(cat) {
    var icons = {
        Food: "🍔", Transport: "🚗", Shopping: "🛍️", Bills: "💡",
        Health: "💊", Entertainment: "🎬", Groceries: "🛒", Donate: "🤲", Others: "💰"
    };
    return icons[cat] || "💰";
}

/***
 * ═══════════════════════════════════════════
 *  1. MONTHLY SUMMARY
 *  Subject: [RupiBook] 📊 Monthly Summary — Jun 2026
 * ═══════════════════════════════════════════
 ***/
function sendMonthlySummary(ss, month) {
    Logger.log("sendMonthlySummary: Preparing monthly digest email for month=" + month);
    var d = getMonthDetails_(ss, month);
    var settings = ss.getSheetByName("Settings");
    var totalLimit = settings ? (Number(settings.getRange("B1").getValue()) || 0) : 0;

    var label = fmtMonthLabel_(month);
    var pct = totalLimit > 0 ? Math.round(d.total / totalLimit * 100) : 0;

    var body = "";
    body += "📊 RupiBook Monthly Summary — " + label + "\n";
    body += "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━\n\n";

    // Overall
    body += "💰 Total Spent: " + CURRENCY + fmt(d.total);
    if(totalLimit > 0) {
        body += " / " + CURRENCY + fmt(totalLimit) + " (" + pct + "%)";
    }
    body += "\n";
    if(totalLimit > 0) {
        var remaining = totalLimit - d.total;
        body += "📊 " + (remaining >= 0 ? "Remaining: " + CURRENCY + fmt(remaining) : "Over budget by: " + CURRENCY + fmt(Math.abs(remaining))) + "\n";
    }
    body += "🔢 Total Transactions: " + d.txCount + "\n\n";

    // Category breakdown
    body += "── Category Breakdown ──\n";
    var cats = Object.keys(d.byCat).sort(function(a, b) { return d.byCat[b] - d.byCat[a]; });
    var overBudgetCats = [];
    var withinBudgetCount = 0;

    cats.forEach(function(c) {
        var spent = d.byCat[c];
        var limit = d.catLimits[c] || 0;
        var line = "  " + catIcon_(c) + " " + c + ": " + CURRENCY + fmt(spent);
        if(limit > 0) {
            var catPct = Math.round(spent / limit * 100);
            line += " / " + CURRENCY + fmt(limit) + " (" + catPct + "%)";
            if(spent > limit) {
                line += " ⚠️ OVER by " + CURRENCY + fmt(spent - limit);
                overBudgetCats.push({ name: c, over: spent - limit });
            } else {
                withinBudgetCount++;
            }
        }
        body += line + "\n";
    });
    body += "\n";

    // Top 5 sources
    var srcArr = Object.keys(d.bySource).map(function(s) {
        return { name: s, amount: d.bySource[s].amount, count: d.bySource[s].count };
    }).sort(function(a, b) { return b.amount - a.amount; }).slice(0, 5);

    if(srcArr.length > 0) {
        body += "── Top 5 Spending Sources ──\n";
        srcArr.forEach(function(s, i) {
            body += "  " + (i + 1) + ". " + s.name + "  " + CURRENCY + fmt(s.amount) + " (" + s.count + " txn)\n";
        });
        body += "\n";
    }

    // Most used payment mode
    var pmArr = Object.keys(d.byPayment).map(function(p) {
        return { name: p, count: d.byPayment[p] };
    }).sort(function(a, b) { return b.count - a.count; });

    if(pmArr.length > 0) {
        body += "── Payment Modes ──\n";
        pmArr.forEach(function(p) {
            body += "  💳 " + p.name + ": " + p.count + " transactions\n";
        });
        body += "\n";
    }

    // Budget analysis
    body += "── Budget Analysis ──\n";
    body += "  ✅ Categories within budget: " + withinBudgetCount + "\n";
    if(overBudgetCats.length > 0) {
        body += "  ⚠️ Categories exceeded: " + overBudgetCats.length + " (";
        body += overBudgetCats.map(function(c) { return c.name + " +" + CURRENCY + fmt(c.over); }).join(", ");
        body += ")\n";
    }
    if(cats.length > 0) {
        body += "  📈 Highest category: " + cats[0] + " (" + CURRENCY + fmt(d.byCat[cats[0]]) + ")\n";
        body += "  📉 Lowest category: " + cats[cats.length - 1] + " (" + CURRENCY + fmt(d.byCat[cats[cats.length - 1]]) + ")\n";
    }
    body += "\n";

    // Monthly comparison — compare with previous month
    var prevParts = month.split("-");
    var prevDate = new Date(Number(prevParts[0]), Number(prevParts[1]) - 2, 1);
    var prevMonth = Utilities.formatDate(prevDate, TIMEZONE, "yyyy-MM");
    var prevData = getMonthDetails_(ss, prevMonth);

    if(prevData.total > 0) {
        var change = d.total - prevData.total;
        var changePct = Math.round(Math.abs(change) / prevData.total * 100);
        body += "── Monthly Comparison ──\n";
        body += "  " + fmtMonthLabel_(prevMonth) + ": " + CURRENCY + fmt(prevData.total);
        body += " → " + fmtMonthLabel_(month) + ": " + CURRENCY + fmt(d.total);
        body += " (" + (change >= 0 ? "+" : "-") + changePct + "%)\n";
    }

    var subject = "[RupiBook] 📊 Monthly Summary — " + label;
    var recipient = Session.getEffectiveUser().getEmail();
    Logger.log("sendMonthlySummary: Sending digest email to " + recipient + ", subject='" + subject + "'");
    MailApp.sendEmail(recipient, subject, body);
}

/***
 * ═══════════════════════════════════════════
 *  2. CATEGORY BUDGET ALERT
 *  Subject: [RupiBook] 🚨 Budget Exceeded — Food — Jun 2026
 * ═══════════════════════════════════════════
 ***/
function sendCategoryBudgetAlert(ss, month, category) {
    Logger.log("sendCategoryBudgetAlert: Preparing category budget exceed alert email for " + category + " (" + month + ")");
    var d = getMonthDetails_(ss, month);
    var label = fmtMonthLabel_(month);
    var catSpent = d.byCat[category] || 0;
    var catLimit = d.catLimits[category] || 0;

    if(catLimit <= 0) return; // No limit set for this category

    var pct = Math.round(catSpent / catLimit * 100);
    var over = catSpent - catLimit;

    var body = "";
    body += "🚨 Budget Exceeded: " + category + " — " + label + "\n";
    body += "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━\n\n";
    body += catIcon_(category) + " " + category + ": " + CURRENCY + fmt(catSpent) + " / " + CURRENCY + fmt(catLimit) + " (" + pct + "%)\n";
    body += "   Exceeded by: " + CURRENCY + fmt(over) + "\n\n";

    // Top sources in this category
    var catExpenses = d.expenses.filter(function(e) { return e.category === category; });
    var catSources = {};
    var highest = null;

    catExpenses.forEach(function(e) {
        if(e.source) {
            if(!catSources[e.source]) catSources[e.source] = { amount: 0, count: 0 };
            catSources[e.source].amount += e.amount;
            catSources[e.source].count++;
        }
        if(!highest || e.amount > highest.amount) {
            highest = e;
        }
    });

    var srcArr = Object.keys(catSources).map(function(s) {
        return { name: s, amount: catSources[s].amount, count: catSources[s].count };
    }).sort(function(a, b) { return b.amount - a.amount; });

    if(srcArr.length > 0) {
        body += "── Top Sources in " + category + " ──\n";
        srcArr.forEach(function(s, i) {
            body += "  " + (i + 1) + ". " + s.name + "  " + CURRENCY + fmt(s.amount) + " (" + s.count + " txn)\n";
        });
        body += "\n";
    }

    // Highest single expense
    if(highest) {
        var hDate = highest.timestamp instanceof Date
            ? Utilities.formatDate(highest.timestamp, TIMEZONE, "MMM dd")
            : String(highest.timestamp).substring(0, 10);
        body += "── Highest Single Expense ──\n";
        body += "  " + CURRENCY + fmt(highest.amount) + " — " + (highest.source || category) + " — " + hDate + "\n\n";
    }

    // Recent expenses in this category (last 10)
    var recent = catExpenses.sort(function(a, b) {
        return new Date(b.timestamp) - new Date(a.timestamp);
    }).slice(0, 10);

    if(recent.length > 0) {
        body += "── Recent " + category + " Expenses ──\n";
        recent.forEach(function(e) {
            var eDate = e.timestamp instanceof Date
                ? Utilities.formatDate(e.timestamp, TIMEZONE, "MMM dd")
                : String(e.timestamp).substring(0, 10);
            body += "  " + eDate + "  " + CURRENCY + fmt(e.amount) + "  " + (e.source || "—") + "\n";
        });
    }

    var subject = "[RupiBook] 🚨 Budget Exceeded — " + category + " — " + label;
    var recipient = Session.getEffectiveUser().getEmail();
    Logger.log("sendCategoryBudgetAlert: Sending budget exceed email to " + recipient + ", subject='" + subject + "'");
    MailApp.sendEmail(recipient, subject, body);
}

/***
 * ═══════════════════════════════════════════
 *  3. OVERALL BUDGET ALERT
 *  Subject: [RupiBook] 🔴 Overall Budget Exceeded — Jun 2026
 * ═══════════════════════════════════════════
 ***/
function sendOverallBudgetAlert(ss, month) {
    Logger.log("sendOverallBudgetAlert: Preparing overall budget exceed alert email for " + month);
    var d = getMonthDetails_(ss, month);
    var settings = ss.getSheetByName("Settings");
    var totalLimit = settings ? (Number(settings.getRange("B1").getValue()) || 0) : 0;

    if(totalLimit <= 0) return;

    var label = fmtMonthLabel_(month);
    var pct = Math.round(d.total / totalLimit * 100);
    var over = d.total - totalLimit;

    var body = "";
    body += "🔴 Overall Budget Exceeded — " + label + "\n";
    body += "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━\n\n";
    body += "💰 Total: " + CURRENCY + fmt(d.total) + " / " + CURRENCY + fmt(totalLimit) + " (" + pct + "%)\n";
    body += "   Exceeded by: " + CURRENCY + fmt(over) + "\n\n";

    // All categories
    body += "── All Categories ──\n";
    var cats = Object.keys(d.byCat).sort(function(a, b) { return d.byCat[b] - d.byCat[a]; });
    var overCats = [];

    cats.forEach(function(c) {
        var spent = d.byCat[c];
        var limit = d.catLimits[c] || 0;
        var line = "  " + catIcon_(c) + " " + c + ": " + CURRENCY + fmt(spent);
        if(limit > 0) {
            line += " / " + CURRENCY + fmt(limit) + " (" + Math.round(spent / limit * 100) + "%)";
            if(spent > limit) {
                line += " ⚠️";
                overCats.push({ name: c, over: spent - limit });
            }
        }
        body += line + "\n";
    });
    body += "\n";

    // Categories over budget
    if(overCats.length > 0) {
        body += "── Categories Over Budget ──\n";
        overCats.forEach(function(c) {
            body += "  " + c.name + ": +" + CURRENCY + fmt(c.over) + "\n";
        });
        body += "\n";
    }

    // Remaining days and daily budget
    var now = new Date();
    var endOfMonth = new Date(now.getFullYear(), now.getMonth() + 1, 0);
    var daysLeft = Math.max(0, endOfMonth.getDate() - now.getDate());

    body += "── Action Needed ──\n";
    body += "  Remaining days in month: " + daysLeft + "\n";
    if(daysLeft > 0) {
        // How much over, so no spending recommended
        body += "  You are " + CURRENCY + fmt(over) + " over budget. Consider reducing non-essential spending.\n";
    }

    var subject = "[RupiBook] 🔴 Overall Budget Exceeded — " + label;
    var recipient = Session.getEffectiveUser().getEmail();
    Logger.log("sendOverallBudgetAlert: Sending overall budget exceed email to " + recipient + ", subject='" + subject + "'");
    MailApp.sendEmail(recipient, subject, body);
}

// ── Named constants ──────────────────────────────────────────
var DEFAULT_BUDGET_LIMIT = 1000; // Default monthly limit (₹) for new categories


// ============================================================
//  syncBudgetCategories                               (NEW — Improvement B)
//  Reads unique categories from Config and appends any that are
//  missing from Budgets. Non-destructive: existing rows untouched.
//  Called by both setupSheets() and refreshDashboard().
// ============================================================
function syncBudgetCategories() {
  Logger.log("syncBudgetCategories: Starting...");
  var ss = SpreadsheetApp.getActiveSpreadsheet();

  var config = ss.getSheetByName("Config");
  var bud    = ss.getSheetByName("Budgets");

  if (!config || !bud) {
    Logger.log("syncBudgetCategories: Config or Budgets sheet missing — skipping.");
    return;
  }

  // Read unique categories from Config (case-insensitive dedup)
  var configLastRow = config.getLastRow();
  var configCategories = [];
  if (configLastRow > 1) {
    var configData = config.getRange(2, 1, configLastRow - 1, 1).getValues();
    var seenLower  = {};
    for (var r = 0; r < configData.length; r++) {
      var cat    = String(configData[r][0]).trim();
      var catKey = cat.toLowerCase();
      if (cat !== "" && !seenLower[catKey]) {
        seenLower[catKey] = true;
        configCategories.push(cat);
      }
    }
  }

  // Read existing Budgets categories (case-insensitive lookup)
  var budLastRow       = bud.getLastRow();
  var existingBudLower = {};
  if (budLastRow > 1) {
    var budData = bud.getRange(2, 1, budLastRow - 1, 1).getValues();
    for (var b = 0; b < budData.length; b++) {
      var budCat = String(budData[b][0]).trim();
      if (budCat !== "") existingBudLower[budCat.toLowerCase()] = true;
    }
  }

  // Append categories missing from Budgets
  var newRows = [];
  for (var c = 0; c < configCategories.length; c++) {
    var key = configCategories[c].toLowerCase();
    if (!existingBudLower[key]) {
      Logger.log("syncBudgetCategories: New budget category → " + configCategories[c]);
      newRows.push([configCategories[c], DEFAULT_BUDGET_LIMIT]);
    }
  }
  if (newRows.length > 0) {
    bud.getRange(bud.getLastRow() + 1, 1, newRows.length, 2).setValues(newRows);
    Logger.log("syncBudgetCategories: Appended " + newRows.length + " new row(s).");
  } else {
    Logger.log("syncBudgetCategories: Nothing to append — Budgets already in sync.");
  }
}

// ============================================================
//  refreshDashboard
//  Syncs Budgets from Config (Improvement B), then fully clears
//  and rebuilds the Dashboard dynamically.
//  Safe to run standalone from the menu — touches no data sheets
//  other than appending missing Budgets rows.
// ============================================================
function refreshDashboard() {
  Logger.log("refreshDashboard: Starting rebuild...");
  var ss = SpreadsheetApp.getActiveSpreadsheet();

  // Improvement B: sync before reading Budgets so new Config categories
  // appear immediately without requiring a full Setup Sheets run.
  syncBudgetCategories();

  var bud = ss.getSheetByName("Budgets");
  if (!bud) {
    SpreadsheetApp.getUi().alert("Budgets sheet not found. Run ⚙️ Setup Sheets first.");
    return;
  }

  // ── Compute dynamic row bounds from Budgets ───────────────────
  var numCategories = Math.max(0, bud.getLastRow() - 1);
  var lastTableRow  = 10 + numCategories;
  Logger.log("refreshDashboard: numCategories=" + numCategories +
             "  lastTableRow=" + lastTableRow);

  // ── Prepare Dashboard sheet ───────────────────────────────────
  var dash = ss.getSheetByName("Dashboard") || ss.insertSheet("Dashboard");
  ss.setActiveSheet(dash);
  ss.moveActiveSheet(1);

  // Guard: ask before wiping an existing Dashboard
  if(dash.getLastRow() > 2) {
      try {
          var ui = SpreadsheetApp.getUi();
          var confirmed = ui.alert(
              "Rebuild Dashboard?",
              "This will reset the Dashboard tab and remove all existing content. Continue?",
              ui.ButtonSet.YES_NO
          );
          if(confirmed !== ui.Button.YES) {
              Logger.log("refreshDashboard: Rebuild cancelled by user.");
              return;
          }
      } catch(e) {
          // getUi() throws when called from a time-based trigger or doGet — proceed silently
          Logger.log("refreshDashboard: running without UI, rebuilding Dashboard.");
      }
  }

  dash.clearContents();
  dash.clearFormats();
  dash.clearConditionalFormatRules();
  var charts = dash.getCharts();
  for (var i = 0; i < charts.length; i++) { dash.removeChart(charts[i]); }

  // ── Title ─────────────────────────────────────────────────────
  dash.getRange("A1").setValue("📊 EXPENSE DASHBOARD");
  dash.getRange("A1:K1")
      .merge()
      .setFontSize(16)
      .setFontWeight("bold")
      .setFontColor(COLOR_DASH_TITLE)
      .setVerticalAlignment("middle");
  dash.setRowHeight(1, 40);

  // ── KPI Summary (A3:B7) ───────────────────────────────────────
  dash.getRange("A3").setValue("This month");
  dash.getRange("B3")
      .setFormula("=EOMONTH(TODAY(),-1)+1")
      .setNumberFormat("mmmm yyyy");

  dash.getRange("A4").setValue("Total spent");
  dash.getRange("B4")
      .setFormula("=SUM(C11:C" + lastTableRow + ")")
      .setNumberFormat("₹#,##0.00");

  dash.getRange("A5").setValue("Monthly limit");
  dash.getRange("B5").setFormula("=Settings!B1").setNumberFormat("₹#,##0.00");

  dash.getRange("A6").setValue("Remaining");
  dash.getRange("B6").setFormula("=B5-B4").setNumberFormat("₹#,##0.00");

  dash.getRange("A7").setValue("% used");
  dash.getRange("B7").setFormula("=B4/B5").setNumberFormat("0.00%");

  dash.getRange("A3:A7").setFontWeight("bold").setBackground(COLOR_KPI_BG);
  dash.getRange("A3:B7").setBorder(
    true, true, true, true, true, true,
    COLOR_BORDER_LIGHT, SpreadsheetApp.BorderStyle.SOLID
  );

  // ── Category Budget Table ─────────────────────────────────────
  if (numCategories > 0) {
    dash.getRange("A10:E10")
        .setValues([["Category", "Limit", "Spent", "Remaining", "% Used"]])
        .setFontWeight("bold")
        .setBackground(COLOR_BLUE_HEADER)
        .setFontColor(COLOR_WHITE)
        .setHorizontalAlignment("center");

    for (var i = 11; i <= lastTableRow; i++) {
      var budgetRow = i - 9;

      dash.getRange("A" + i).setFormula("=Budgets!A" + budgetRow);

      dash.getRange("B" + i)
          .setFormula("=Budgets!B" + budgetRow)
          .setNumberFormat("₹#,##0.00");

      // Improvement A: TEXT(TODAY(),"yyyy-MM") instead of TEXT($B$3,"yyyy-MM")
      // so that editing the display cell B3 can never silently break SUMIFS.
      dash.getRange("C" + i)
          .setFormula(
            '=SUMIFS(Expenses!$B:$B,' +
            'Expenses!$C:$C,$A' + i + ',' +
            'Expenses!$H:$H,TEXT(TODAY(),"yyyy-MM"))'
          )
          .setNumberFormat("₹#,##0.00");

      dash.getRange("D" + i)
          .setFormula("=B" + i + "-C" + i)
          .setNumberFormat("₹#,##0.00");

      dash.getRange("E" + i)
          .setFormula("=IF(B" + i + ">0,C" + i + "/B" + i + ",0)")
          .setNumberFormat("0.00%");
    }

    dash.getRange("A11:E" + lastTableRow).setBorder(
      true, true, true, true, true, true,
      COLOR_BORDER_TABLE, SpreadsheetApp.BorderStyle.SOLID
    );

    // Conditional formatting over the dynamic E range
    var threshold = (Number(
      ss.getSheetByName("Settings").getRange("B2").getValue()
    ) || 75) / 100;

    var rangeE    = dash.getRange("E11:E" + lastTableRow);
    var ruleGreen = SpreadsheetApp.newConditionalFormatRule()
      .whenNumberLessThan(threshold)
      .setBackground(COLOR_CF_GREEN_BG).setFontColor(COLOR_CF_GREEN_FG)
      .setRanges([rangeE]).build();
    var ruleRed   = SpreadsheetApp.newConditionalFormatRule()
      .whenNumberGreaterThanOrEqualTo(threshold)
      .setBackground(COLOR_CF_RED_BG).setFontColor(COLOR_CF_RED_FG)
      .setRanges([rangeE]).build();
    dash.setConditionalFormatRules([ruleGreen, ruleRed]);
  }

  // ── Pivot: Category vs Spent (G:H) ───────────────────────────
  dash.getRange("G10").setFormula(
    '=QUERY(A10:C' + lastTableRow + ',' +
    '"select A, C where C > 0 order by A asc",1)'
  );
  dash.getRange("G10:H10")
      .setFontWeight("bold")
      .setBackground(COLOR_GREEN_HEADER)
      .setFontColor(COLOR_WHITE);

  // ── Pivot: Monthly Spending (J:L) ─────────────────────────────
  dash.getRange("J10").setValue("Month");
  dash.getRange("K10").setValue("Spent");
  dash.getRange("L10").setValue("Budget");

  dash.getRange("J11").setFormula(
    '=QUERY(Expenses!A:H,' +
    '"select H, sum(B) where H is not null ' +
    'group by H order by H ' +
    'label H \'\', sum(B) \'\'",0)'
  );
  dash.getRange("L11").setFormula(
    '=ARRAYFORMULA(IF(J11:J="","",Settings!B1))'
  );

  dash.getRange("J10:L10")
      .setFontWeight("bold")
      .setBackground(COLOR_YELLOW_HEADER)
      .setFontColor(COLOR_WHITE);

  // ── Compute lastMonthRow for formatting + Combo chart ─────────
  // Improvement C: scan ceiling = max(dash.getLastRow()+50, expLastRow)
  // instead of dash.getMaxRows(), avoiding walks over thousands of empty rows.
  SpreadsheetApp.flush();

  var exp         = ss.getSheetByName("Expenses");
  var expLastRow  = exp ? exp.getLastRow() : 11;
  var scanCeiling = Math.max(dash.getLastRow() + 50, expLastRow);

  var fullJData    = dash.getRange("J11:J" + scanCeiling).getValues();
  var lastMonthRow = 10;
  for (var j = fullJData.length - 1; j >= 0; j--) {
    if (fullJData[j][0] !== "") { lastMonthRow = 11 + j; break; }
  }
  if (lastMonthRow < 11) lastMonthRow = 11;
  Logger.log("refreshDashboard: lastMonthRow=" + lastMonthRow +
             "  (scanCeiling=" + scanCeiling + ")");

  dash.getRange("K11:K" + lastMonthRow).setNumberFormat("₹#,##0.00");
  dash.getRange("J11:J" + lastMonthRow).setNumberFormat("yyyy-mm");

  // ── Compute lastPieRow for Pie chart ──────────────────────────
  // Improvement C: same bounded ceiling for G-column scan.
  SpreadsheetApp.flush();

  var fullGData  = dash.getRange("G11:G" + scanCeiling).getValues();
  var lastPieRow = 10;
  for (var g = fullGData.length - 1; g >= 0; g--) {
    if (fullGData[g][0] !== "") { lastPieRow = 11 + g; break; }
  }
  if (lastPieRow < 11) lastPieRow = 11;
  Logger.log("refreshDashboard: lastPieRow=" + lastPieRow);

  // ── Charts ────────────────────────────────────────────────────
  var staleCharts = dash.getCharts();
  for (var k = 0; k < staleCharts.length; k++) { dash.removeChart(staleCharts[k]); }

  if (numCategories > 0) {
    try {
      var pieChart = dash.newChart()
        .setChartType(Charts.ChartType.PIE)
        .addRange(dash.getRange("G10:H" + lastPieRow))
        .setPosition(20, 1, 0, 0)
        .setOption("title", "Expenses by Category")
        .setOption("is3D", true)
        .build();
      dash.insertChart(pieChart);
      Logger.log("refreshDashboard: Pie chart inserted (G10:H" + lastPieRow + ").");
    } catch (err) {
      Logger.log("refreshDashboard: Pie chart failed — " + err.message);
    }

    try {
      var budgetChart = dash.newChart()
        .setChartType(Charts.ChartType.BAR)
        .addRange(dash.getRange("A10:A" + lastTableRow))
        .addRange(dash.getRange("B10:B" + lastTableRow))
        .addRange(dash.getRange("C10:C" + lastTableRow))
        .setPosition(40, 1, 0, 0)
        .setOption("title", "Current Month: Budget vs Actual")
        .setOption("legend", { position: "top" })
        .setOption("colors", [COLOR_GREEN_HEADER, COLOR_RED_HEADER])
        .setOption("hAxis", { title: "Amount" })
        .build();
      dash.insertChart(budgetChart);
      Logger.log("refreshDashboard: Bar chart inserted.");
    } catch (err) {
      Logger.log("refreshDashboard: Bar chart failed — " + err.message);
    }
  }

  try {
    var comboChart = dash.newChart()
      .setChartType(Charts.ChartType.COMBO)
      .addRange(dash.getRange("J10:L" + lastMonthRow))
      .setPosition(20, 10, 0, 0)
      .setOption("title", "Monthly Spending vs Budget")
      .setOption("seriesType", "bars")
      .setOption("series", { 1: { type: "line" } })
      .build();
    dash.insertChart(comboChart);
    Logger.log("refreshDashboard: Combo chart inserted (J10:L" + lastMonthRow + ").");
  } catch (err) {
    Logger.log("refreshDashboard: Combo chart failed — " + err.message);
  }

  Logger.log("refreshDashboard: Complete.");
}


// ============================================================
//  migrateTimestamps
//  One-time migration: converts string timestamps in Expenses
//  column A from "yyyy-MM-dd HH:mm:ss" to real Date objects.
//  Safe to re-run — already-Date rows are skipped.
// ============================================================
function migrateTimestamps() {
  Logger.log("migrateTimestamps: Starting...");
  var ss  = SpreadsheetApp.getActiveSpreadsheet();
  var exp = ss.getSheetByName("Expenses");

  if (!exp) {
    SpreadsheetApp.getUi().alert("Expenses sheet not found.");
    return;
  }

  var lastRow = exp.getLastRow();
  if (lastRow < 2) {
    SpreadsheetApp.getUi().alert("No expense rows to migrate.");
    return;
  }

  var tsRange  = exp.getRange(2, 1, lastRow - 1, 1);
  var tsValues = tsRange.getValues();
  var converted = 0, skipped = 0, failed = 0;

  for (var i = 0; i < tsValues.length; i++) {
    var cell = tsValues[i][0];

    if (cell instanceof Date && !isNaN(cell.getTime())) { skipped++; continue; }

    if (typeof cell === "string" && cell.trim() !== "") {
      var str   = cell.trim();
      var parts = str.split(" ");
      if (parts.length === 2) {
        var dateParts = parts[0].split("-");
        var timeParts = parts[1].split(":");
        if (dateParts.length === 3 && timeParts.length === 3) {
          var y   = parseInt(dateParts[0], 10);
          var mon = parseInt(dateParts[1], 10) - 1;
          var d   = parseInt(dateParts[2], 10);
          var h   = parseInt(timeParts[0], 10);
          var min = parseInt(timeParts[1], 10);
          var s   = parseInt(timeParts[2], 10);
          var parsed = new Date(y, mon, d, h, min, s);
          if (!isNaN(parsed.getTime())) {
            tsValues[i][0] = parsed;
            converted++;
            continue;
          }
        }
      }
      Logger.log("migrateTimestamps: Row " + (i + 2) + " unparseable: " + cell);
      failed++;
    } else {
      skipped++;
    }
  }

  if (converted > 0) {
    tsRange.setValues(tsValues);
    exp.getRange("A:A").setNumberFormat("dd-MM-yyyy HH:mm:ss");
  }

  Logger.log("migrateTimestamps: converted=" + converted +
             "  skipped=" + skipped + "  failed=" + failed);

  SpreadsheetApp.getUi().alert(
    "Timestamp migration complete!\n\n" +
    "✅ Converted : " + converted + " rows\n" +
    "⏭  Skipped   : " + skipped + " rows (already Date or empty)\n" +
    (failed > 0 ? "❌ Failed    : " + failed + " rows (check Logs)\n" : "") +
    "\nRun ⚙️ Setup Sheets next to rebuild the Dashboard."
  );
}

// Indian number formatting: 1234567 → 12,34,567
function fmt(n) {
    n = Math.round(Number(n) || 0);
    var s = String(n);
    var last3 = s.slice(-3);
    var rest = s.slice(0, -3);
    if(rest !== "") last3 = "," + last3;
    return rest.replace(/\B(?=(\d{2})+(?!\d))/g, ",") + last3;
}

// Returns category→source mapping and payment modes for the PWA dropdowns
function getConfigData() {
    Logger.log("getConfigData: Fetching dropdown categories, mapping, and payment modes...");
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName("Config");

    var mapping = {};
    var categories = [];

    if(sheet) {
        var data = sheet.getDataRange().getValues();
        for(var i = 1; i < data.length; i++) {
            var category = String(data[i][0] || "").trim();
            var source = String(data[i][1] || "").trim();

            if(!category || !source) continue;

            if(!mapping[category]) {
                mapping[category] = [];
                categories.push(category);
            }

            mapping[category].push(source);
        }
    }

    var pmSheet = ss.getSheetByName("PaymentModes");
    var paymentModes = [];

    if(pmSheet) {
        var pmData = pmSheet.getDataRange().getValues();
        for(var j = 1; j < pmData.length; j++) {
            var mode = String(pmData[j][0] || "").trim();
            if(mode) paymentModes.push(mode);
        }
    }

    Logger.log("getConfigData: Loaded " + categories.length + " categories and " + paymentModes.length + " payment modes.");
    return {
        categories: categories,
        mapping: mapping,
        paymentModes: paymentModes
    };
}

// Returns current month's summary for the dashboard
function getDashboardData() {
    Logger.log("getDashboardData: Preparing statistics for the dashboard...");
    var ss = SpreadsheetApp.getActiveSpreadsheet();

    var expenses = ss.getSheetByName("Expenses");
    var settings = ss.getSheetByName("Settings");

    var month = Utilities.formatDate(
        new Date(),
        TIMEZONE,
        "yyyy-MM"
    );

    var total = 0;
    var categoryTotals = {};

    var rows = expenses.getDataRange().getValues();

    for(var i = 1; i < rows.length; i++) {
        if(!rows[i][0] || rows[i][1] === "") continue;

        var rowMonth = rows[i][7];

        if(rowMonth instanceof Date) {
            rowMonth = Utilities.formatDate(
                rowMonth,
                TIMEZONE,
                "yyyy-MM"
            );
        }

        if(String(rowMonth) !== month) continue;

        var amount = Number(rows[i][1]) || 0;
        var category = rows[i][2] || "Others";

        total += amount;

        categoryTotals[category] =
            (categoryTotals[category] || 0) + amount;
    }

    var limit = settings
        ? (Number(settings.getRange("B1").getValue()) || 0)
        : 0;

    // Read category budgets for the dashboard overshoot chart
    var categoryBudgets = {};
    var budgets = ss.getSheetByName("Budgets");
    if(budgets) {
        var bData = budgets.getDataRange().getValues();
        for(var j = 1; j < bData.length; j++) {
            var cn = String(bData[j][0] || "").trim();
            if(cn) categoryBudgets[cn] = Number(bData[j][1]) || 0;
        }
    }

    Logger.log("getDashboardData: Compiled metrics for " + month + ". Spent=" + total + ", limit=" + limit);
    return {
        month: month,
        totalSpent: total,
        monthlyLimit: limit,
        remaining: limit - total,
        budgetPercent: limit > 0 ?
            Math.round((total / limit) * 100) : 0,
        categoryTotals: categoryTotals,
        categoryBudgets: categoryBudgets
    };
}

// Returns distinct categories, sources, and tags for filter chips
function getFilterValues() {
    Logger.log("getFilterValues: Aggregating unique categories, sources, and tags for client filters...");
    var sheet =
        SpreadsheetApp.getActiveSpreadsheet()
        .getSheetByName("Expenses");

    var rows = sheet.getDataRange().getValues();

    var categories = {};
    var sources = {};
    var tags = {};

    for(var i = 1; i < rows.length; i++) {

        var category = String(rows[i][2] || "").trim();
        var source = String(rows[i][3] || "").trim();
        var tagString = String(rows[i][5] || "").trim();

        if(category) categories[category] = true;
        if(source) sources[source] = true;

        if(tagString) {

            tagString
                .split(/\s+/)
                .forEach(function(tag) {

                    tag = tag.trim();

                    if(!tag) return;

                    tags[tag] = true;
                });
        }
    }

    var keysCat = Object.keys(categories);
    var keysSrc = Object.keys(sources);
    var keysTag = Object.keys(tags);
    Logger.log("getFilterValues: Found " + keysCat.length + " categories, " + keysSrc.length + " sources, and " + keysTag.length + " unique tags.");
    return {
        categories: keysCat.sort(),
        sources: keysSrc.sort(),
        tags: keysTag.sort()
    };
}

/***
 * ───────── SEARCH ─────────
 * Server-side filtered search. Accepts query parameters:
 *   q         — free-text search (matches category, source, tags, comments)
 *   category  — exact category match
 *   source    — exact source match
 *   tag       — matches if the tag appears in the tags field
 *   from      — date string (yyyy-MM-dd), inclusive lower bound
 *   to        — date string (yyyy-MM-dd), inclusive upper bound
 *   limit     — max results to return (default 50)
 *
 * Returns the most recent matching expenses (newest first).
 ***/
function searchExpenses(params) {
    Logger.log("searchExpenses: Querying expenses. Parameters: " + JSON.stringify(params));
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Expenses");
    var rows = sheet.getDataRange().getValues();

    var q = String(params.q || "").trim().toLowerCase();
    var category = String(params.category || "").trim();
    var source = String(params.source || "").trim();
    var tag = String(params.tag || "").trim().toLowerCase();
    var fromStr = String(params.from || "").trim();
    var toStr = String(params.to || "").trim();
    var limit = Number(params.limit) || 500;

    var results = [];

    // Iterate newest-first
    for(var i = rows.length - 1; i >= 1 && results.length < limit; i--) {
        var rowTs = rows[i][0];
        var rowAmount = Number(rows[i][1]) || 0;
        var rowCategory = String(rows[i][2] || "").trim();
        var rowSource = String(rows[i][3] || "").trim();
        var rowTags = String(rows[i][5] || "").trim();
        var rowComments = String(rows[i][6] || "").trim();

        // Skip blank rows
        if(!rowTs || rowAmount === "") continue;

        // Category filter
        if(category && rowCategory !== category) continue;

        // Source filter
        if(source && rowSource !== source) continue;

        // Tag filter
        if(tag) {
            var tagsLower = rowTags.toLowerCase();
            if(tagsLower.indexOf(tag) === -1) continue;
        }

        // Date range filter — compare in IST so early-morning entries (12am–5:30am IST)
        // are not excluded due to the UTC date being one day earlier than the IST date.
        // This matches how getDashboardData() filters using the pre-computed Month column.
        var rowDate = (rowTs instanceof Date) ? rowTs : new Date(rowTs);
        var rowDateStr = "";
        if (rowDate && !isNaN(rowDate.getTime())) {
            rowDateStr = Utilities.formatDate(rowDate, TIMEZONE, "yyyy-MM-dd");
        } else {
            var tsStr = String(rowTs || "").trim();
            if (tsStr.match(/^\d{4}-\d{2}-\d{2}/)) {
                rowDateStr = tsStr.substring(0, 10);
            }
        }
        if(fromStr && rowDateStr < fromStr) continue;
        if(toStr   && rowDateStr > toStr)   continue;

        // Free-text search (matches category, source, tags, comments)
        if(q) {
            var haystack = (rowCategory + " " + rowSource + " " + rowTags + " " + rowComments).toLowerCase();
            if(haystack.indexOf(q) === -1) continue;
        }

        results.push({
            timestamp: rows[i][0],
            amount: rows[i][1],
            category: rowCategory,
            source: rowSource,
            paymentMode: String(rows[i][4] || "").trim(),
            tags: rowTags,
            comments: rowComments
        });
    }

    Logger.log("searchExpenses: Found " + results.length + " matching expense(s)");
    return results;
}
