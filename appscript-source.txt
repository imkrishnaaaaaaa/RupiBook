/*** ───────── CONFIG ───────── ***/
// Change these to match your locale
var TIMEZONE = "Asia/Kolkata";
var CURRENCY = "₹";

/*** App version — bump this on every deploy.
     The PWA compares this against its own RUPIBOOK_UI_VERSION (in js/version.js).
     If they differ, an "update required" banner is shown to the user. ***/
var APPSCRIPT_VERSION = "1.2.3";

/***
 * ───────── doPost: receives expense data or actions from the PWA / Shortcut ─────────
 * 
 * Expected JSON body for logging:
 *   { amount, category, source, paymentMode, tags, comments }
 *
 * Special actions:
 *   { action: "undo" }  →  removes the last logged expense
 ***/
function doPost(e) {
    var lock = LockService.getScriptLock();
    lock.tryLock(10000);
    try {
        var ss = SpreadsheetApp.getActiveSpreadsheet();
        var p = JSON.parse(e.postData.contents);

        if(p.action === "undo") {
            return undoLast(ss);
        }

        var sheet = ss.getSheetByName("Expenses");
        var now = new Date();
        var timestamp = Utilities.formatDate(now, TIMEZONE, "yyyy-MM-dd HH:mm:ss");
        var month = Utilities.formatDate(now, TIMEZONE, "yyyy-MM");

        var amount = Number(p.amount) || 0;
        var category = p.category || "Others";
        var source = p.source || "";
        var paymentMode = p.paymentMode || "";
        var tags = p.tags || "";
        var comments = p.comments || "";

        // Expenses row: Timestamp | Amount | Category | Source | Payment Mode | Tags | Comments | Month
        sheet.appendRow([timestamp, amount, category, source, paymentMode, tags, comments, month]);

        var msg = checkBudgets(ss, month, category);
        return json({
            status: "ok",
            message: msg
        });

    } catch (err) {
        return json({
            status: "error",
            message: "Error: " + err
        });
    } finally {
        lock.releaseLock();
    }
}

/***
 * ───────── doGet: REST-like API, routed by ?action= parameter ─────────
 *
 * Actions:
 *   version   → { version }                         — for PWA version sync
 *   config    → { categories, mapping, paymentModes } — dropdown data
 *   dashboard → { month, totalSpent, ... }           — current month summary
 *   recent    → [ last 20 expenses ]                 — for dashboard list
 *   analytics → { breakdowns + recentExpenses }      — for analytics page
 *   filters   → { categories, sources, tags }        — distinct filter values
 *   search    → [ filtered expenses, max 50 ]        — server-side search
 *   (none)    → health check
 ***/
function doGet(e) {

    var action = e.parameter.action || "";

    if(action === "version") {
        return json({
            version: APPSCRIPT_VERSION
        });
    }

    if(action === "config") {
        return json(getConfigData());
    }

    if(action === "dashboard") {
        return json(getDashboardData());
    }

    if(action === "recent") {
        return json(getRecentExpenses());
    }

    if(action === "analytics") {
        return json(getAnalyticsData());
    }

    if(action === "filters") {
        return json(getFilterValues());
    }

    if(action === "search") {
        return json(searchExpenses(e.parameter));
    }

    // Default health check
    return json({
        status: "ok",
        message: "RupiBook API Running",
        version: APPSCRIPT_VERSION
    });
}

/***
 * ───────── BUDGET CHECK ─────────
 * Called after every expense log. Compares this month's spending against:
 *   - Per-category limit from the Budgets sheet
 *   - Overall monthly limit from Settings!B1
 * Returns a human-readable alert string (shown as a notification on the phone).
 * Also triggers email alerts when budgets are exceeded (with duplicate prevention).
 ***/
function checkBudgets(ss, month, category) {
    var settings = ss.getSheetByName("Settings");

    // Default 80% threshold if Settings sheet is missing
    var threshold = 0.8;
    var totalLimit = 0;

    if(settings) {
        totalLimit = Number(settings.getRange("B1").getValue()) || 0;
        // Settings!B2 holds the alert threshold as a percentage (e.g. 75 = 75%)
        var t = Number(settings.getRange("B2").getValue());
        if(t > 0) {
            threshold = t / 100;
        }
    }

    // Sum all expenses for this month
    var data = ss.getSheetByName("Expenses").getDataRange().getValues();
    var totalSpent = 0;
    var catSpent = 0;

    for(var i = 1; i < data.length; i++) {
        if(!data[i][0] || data[i][1] === "") {
            continue;
        }

        var rowMonth = data[i][7]; // Column H = Month

        if(rowMonth instanceof Date) {
            rowMonth = Utilities.formatDate(rowMonth, TIMEZONE, "yyyy-MM");
        }

        rowMonth = String(rowMonth || "").trim();

        if(rowMonth !== month) {
            continue;
        }

        var amt = Number(data[i][1]) || 0; // Column B = Amount
        totalSpent += amt;

        if(String(data[i][2] || "").trim() === category) { // Column C = Category
            catSpent += amt;
        }
    }

    // Look up this category's limit from the Budgets sheet
    var catLimit = 0;
    var budgets = ss.getSheetByName("Budgets");

    if(budgets) {
        var b = budgets.getDataRange().getValues();
        for(var j = 1; j < b.length; j++) {
            if(String(b[j][0] || "").trim() === String(category || "").trim()) {
                catLimit = Number(b[j][1]) || 0;
                break;
            }
        }
    }

    // Build alert messages
    var alerts = [];

    if(catLimit > 0 && catSpent >= threshold * catLimit) {
        alerts.push(`⚠️ ${category}: ${CURRENCY}${fmt(catSpent)} of ${CURRENCY}${fmt(catLimit)} (${Math.round((catSpent / catLimit) * 100)}%). ${CURRENCY}${fmt(catLimit - catSpent)} left.`);
    }

    if(totalLimit > 0 && totalSpent >= threshold * totalLimit) {
        alerts.push(`🚨 Overall: ${CURRENCY}${fmt(totalSpent)} of ${CURRENCY}${fmt(totalLimit)} (${Math.round((totalSpent / totalLimit) * 100)}%). ${CURRENCY}${fmt(totalLimit - totalSpent)} left.`);
    }

    // ── Email alerts when budgets EXCEED 100% ──
    // Category budget exceeded
    if(catLimit > 0 && catSpent > catLimit) {
        trySendCategoryAlert_(ss, month, category);
    }
    // Overall budget exceeded
    if(totalLimit > 0 && totalSpent > totalLimit) {
        trySendOverallAlert_(ss, month);
    }

    if(alerts.length === 0) {
        return `✅ Logged · ${category}. Month so far: ${CURRENCY}${fmt(totalSpent)}${totalLimit > 0 ? ` / ${CURRENCY}${fmt(totalLimit)}` : ""}.`;
    }

    return alerts.join("\n");
}

/***
 * ───────── Alert deduplication helpers ─────────
 * Uses PropertiesService to track which alerts have been sent.
 * Category alerts: sent once on exceed, then re-sent every 10 days with current status.
 * Overall alerts: sent once on exceed, then re-sent every 10 days.
 ***/
function trySendCategoryAlert_(ss, month, category) {
    var props = PropertiesService.getScriptProperties();
    var key = "alert_cat_" + category + "_" + month;
    var last = props.getProperty(key);
    var now = new Date().getTime();

    if(last) {
        var daysSince = (now - Number(last)) / (1000 * 60 * 60 * 24);
        if(daysSince < 10) return; // Already sent within last 10 days
    }

    try {
        sendCategoryBudgetAlert(ss, month, category);
        props.setProperty(key, String(now));
    } catch(e) {
        Logger.log("Category alert email failed: " + e);
    }
}

function trySendOverallAlert_(ss, month) {
    var props = PropertiesService.getScriptProperties();
    var key = "alert_overall_" + month;
    var last = props.getProperty(key);
    var now = new Date().getTime();

    if(last) {
        var daysSince = (now - Number(last)) / (1000 * 60 * 60 * 24);
        if(daysSince < 10) return;
    }

    try {
        sendOverallBudgetAlert(ss, month);
        props.setProperty(key, String(now));
    } catch(e) {
        Logger.log("Overall alert email failed: " + e);
    }
}

/*** ───────── UNDO LAST ENTRY ───────── ***/
function undoLast(ss) {
    var sheet = ss.getSheetByName("Expenses");
    var last = sheet.getLastRow();
    if(last > 1) {
        var row = sheet.getRange(last, 1, 1, 8).getValues()[0];
        sheet.deleteRow(last);
        return json({
            status: "ok",
            message: "↩️ Removed: " + CURRENCY + fmt(row[1]) + " · " + row[2]
        });
    }
    return json({
        status: "ok",
        message: "Nothing to undo."
    });
}

/***
 * ───────── AUTOPAY ─────────
 * Set up a daily time-driven trigger for this function:
 *   Triggers → + Add Trigger → logAutopay → Time-driven → Day timer
 *
 * It checks every row in the Autopay sheet. If Active=true and
 * Day of Month matches today, it appends a row to Expenses.
 ***/
function logAutopay() {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var ap = ss.getSheetByName("Autopay");
    var exp = ss.getSheetByName("Expenses");
    if(!ap) return;
    var today = new Date();
    var dayNum = Number(Utilities.formatDate(today, TIMEZONE, "d"));
    var timestamp = Utilities.formatDate(today, TIMEZONE, "yyyy-MM-dd HH:mm:ss");
    var month = Utilities.formatDate(today, TIMEZONE, "yyyy-MM");
    var rows = ap.getDataRange().getValues();
    for(var i = 1; i < rows.length; i++) {
        var name = rows[i][0],
            amount = rows[i][1],
            category = rows[i][2];
        var day = Number(rows[i][3]),
            active = rows[i][4];
        if(active === true && day === dayNum) {
            exp.appendRow([timestamp, amount, category, name, "Autopay", "autopay", "Auto-logged: " + name, month]);
        }
    }
}

/***
 * ───────── EMAIL SYSTEM ─────────
 *
 * Three types of emails, each with distinct subject patterns for Gmail filtering:
 *   1. [RupiBook] 📊 Monthly Summary — Jun 2026       → 1st of each month (prev month recap)
 *   2. [RupiBook] 🚨 Budget Exceeded — Food — Jun 2026 → when a category exceeds its budget
 *   3. [RupiBook] 🔴 Overall Budget Exceeded — Jun 2026 → when total exceeds monthly limit
 *
 * Trigger setup: create a DAILY time-driven trigger for `dailyTrigger`.
 * Budget alerts (2 & 3) are also fired from checkBudgets() on each expense,
 * with dedup via PropertiesService (re-sent every 10 days max).
 ***/

/*** ── Daily trigger orchestrator ──
 *  Fires monthly summary on the 1st of each month (for the previous month).
 *  Kept as the single function to attach a daily trigger to.
 ***/
function dailyTrigger() {
    var today = new Date();
    var dayOfMonth = Number(Utilities.formatDate(today, TIMEZONE, "d"));

    // On the 1st, send last month's summary
    if(dayOfMonth === 1) {
        var lastMonth = new Date(today.getFullYear(), today.getMonth() - 1, 1);
        var month = Utilities.formatDate(lastMonth, TIMEZONE, "yyyy-MM");
        var ss = SpreadsheetApp.getActiveSpreadsheet();
        try {
            sendMonthlySummary(ss, month);
        } catch(e) {
            Logger.log("Monthly summary email failed: " + e);
        }
    }
}

/*** Backward compat: old trigger name still works ***/
function sendDailyDigest() { dailyTrigger(); }

/***
 * ── Helper: get month expenses with full details ──
 * Returns { total, byCat, bySource, byPayment, expenses[], catLimits }
 ***/
function getMonthDetails_(ss, month) {
    var data = ss.getSheetByName("Expenses").getDataRange().getValues();
    var total = 0;
    var byCat = {};      // category → amount
    var bySource = {};   // source → { amount, count }
    var byPayment = {};  // paymentMode → count
    var expenses = [];   // full expense objects
    var txCount = 0;

    for(var i = 1; i < data.length; i++) {
        if(!data[i][0] || data[i][1] === "") continue;

        var rowMonth = data[i][7];
        if(rowMonth instanceof Date) {
            rowMonth = Utilities.formatDate(rowMonth, TIMEZONE, "yyyy-MM");
        }
        rowMonth = String(rowMonth || "").trim();
        if(rowMonth !== month) continue;

        var amt = Number(data[i][1]) || 0;
        var cat = String(data[i][2] || "Others").trim();
        var src = String(data[i][3] || "").trim();
        var pm = String(data[i][4] || "").trim();

        total += amt;
        txCount++;
        byCat[cat] = (byCat[cat] || 0) + amt;

        if(src) {
            if(!bySource[src]) bySource[src] = { amount: 0, count: 0 };
            bySource[src].amount += amt;
            bySource[src].count++;
        }

        if(pm) {
            byPayment[pm] = (byPayment[pm] || 0) + 1;
        }

        expenses.push({
            timestamp: data[i][0],
            amount: amt,
            category: cat,
            source: src,
            paymentMode: pm,
            tags: String(data[i][5] || "").trim(),
            comments: String(data[i][6] || "").trim()
        });
    }

    // Get category budgets
    var catLimits = {};
    var budgets = ss.getSheetByName("Budgets");
    if(budgets) {
        var b = budgets.getDataRange().getValues();
        for(var j = 1; j < b.length; j++) {
            var cn = String(b[j][0] || "").trim();
            if(cn) catLimits[cn] = Number(b[j][1]) || 0;
        }
    }

    return {
        total: total,
        txCount: txCount,
        byCat: byCat,
        bySource: bySource,
        byPayment: byPayment,
        expenses: expenses,
        catLimits: catLimits
    };
}

/*** ── Helper: format month label (yyyy-MM → "Jun 2026") ── ***/
function fmtMonthLabel_(month) {
    var parts = month.split("-");
    var names = ["Jan","Feb","Mar","Apr","May","Jun","Jul","Aug","Sep","Oct","Nov","Dec"];
    return names[Number(parts[1]) - 1] + " " + parts[0];
}

/*** ── Helper: category icon ── ***/
function catIcon_(cat) {
    var icons = {
        Food: "🍔", Transport: "🚗", Shopping: "🛍️", Bills: "💡",
        Health: "💊", Entertainment: "🎬", Groceries: "🛒", Donate: "🤲", Others: "💰"
    };
    return icons[cat] || "💰";
}

/***
 * ═══════════════════════════════════════════
 *  1. MONTHLY SUMMARY
 *  Subject: [RupiBook] 📊 Monthly Summary — Jun 2026
 * ═══════════════════════════════════════════
 ***/
function sendMonthlySummary(ss, month) {
    var d = getMonthDetails_(ss, month);
    var settings = ss.getSheetByName("Settings");
    var totalLimit = settings ? (Number(settings.getRange("B1").getValue()) || 0) : 0;

    var label = fmtMonthLabel_(month);
    var pct = totalLimit > 0 ? Math.round(d.total / totalLimit * 100) : 0;

    var body = "";
    body += "📊 RupiBook Monthly Summary — " + label + "\n";
    body += "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━\n\n";

    // Overall
    body += "💰 Total Spent: " + CURRENCY + fmt(d.total);
    if(totalLimit > 0) {
        body += " / " + CURRENCY + fmt(totalLimit) + " (" + pct + "%)";
    }
    body += "\n";
    if(totalLimit > 0) {
        var remaining = totalLimit - d.total;
        body += "📊 " + (remaining >= 0 ? "Remaining: " + CURRENCY + fmt(remaining) : "Over budget by: " + CURRENCY + fmt(Math.abs(remaining))) + "\n";
    }
    body += "🔢 Total Transactions: " + d.txCount + "\n\n";

    // Category breakdown
    body += "── Category Breakdown ──\n";
    var cats = Object.keys(d.byCat).sort(function(a, b) { return d.byCat[b] - d.byCat[a]; });
    var overBudgetCats = [];
    var withinBudgetCount = 0;

    cats.forEach(function(c) {
        var spent = d.byCat[c];
        var limit = d.catLimits[c] || 0;
        var line = "  " + catIcon_(c) + " " + c + ": " + CURRENCY + fmt(spent);
        if(limit > 0) {
            var catPct = Math.round(spent / limit * 100);
            line += " / " + CURRENCY + fmt(limit) + " (" + catPct + "%)";
            if(spent > limit) {
                line += " ⚠️ OVER by " + CURRENCY + fmt(spent - limit);
                overBudgetCats.push({ name: c, over: spent - limit });
            } else {
                withinBudgetCount++;
            }
        }
        body += line + "\n";
    });
    body += "\n";

    // Top 5 sources
    var srcArr = Object.keys(d.bySource).map(function(s) {
        return { name: s, amount: d.bySource[s].amount, count: d.bySource[s].count };
    }).sort(function(a, b) { return b.amount - a.amount; }).slice(0, 5);

    if(srcArr.length > 0) {
        body += "── Top 5 Spending Sources ──\n";
        srcArr.forEach(function(s, i) {
            body += "  " + (i + 1) + ". " + s.name + "  " + CURRENCY + fmt(s.amount) + " (" + s.count + " txn)\n";
        });
        body += "\n";
    }

    // Most used payment mode
    var pmArr = Object.keys(d.byPayment).map(function(p) {
        return { name: p, count: d.byPayment[p] };
    }).sort(function(a, b) { return b.count - a.count; });

    if(pmArr.length > 0) {
        body += "── Payment Modes ──\n";
        pmArr.forEach(function(p) {
            body += "  💳 " + p.name + ": " + p.count + " transactions\n";
        });
        body += "\n";
    }

    // Budget analysis
    body += "── Budget Analysis ──\n";
    body += "  ✅ Categories within budget: " + withinBudgetCount + "\n";
    if(overBudgetCats.length > 0) {
        body += "  ⚠️ Categories exceeded: " + overBudgetCats.length + " (";
        body += overBudgetCats.map(function(c) { return c.name + " +" + CURRENCY + fmt(c.over); }).join(", ");
        body += ")\n";
    }
    if(cats.length > 0) {
        body += "  📈 Highest category: " + cats[0] + " (" + CURRENCY + fmt(d.byCat[cats[0]]) + ")\n";
        body += "  📉 Lowest category: " + cats[cats.length - 1] + " (" + CURRENCY + fmt(d.byCat[cats[cats.length - 1]]) + ")\n";
    }
    body += "\n";

    // Monthly comparison — compare with previous month
    var prevParts = month.split("-");
    var prevDate = new Date(Number(prevParts[0]), Number(prevParts[1]) - 2, 1);
    var prevMonth = Utilities.formatDate(prevDate, TIMEZONE, "yyyy-MM");
    var prevData = getMonthDetails_(ss, prevMonth);

    if(prevData.total > 0) {
        var change = d.total - prevData.total;
        var changePct = Math.round(Math.abs(change) / prevData.total * 100);
        body += "── Monthly Comparison ──\n";
        body += "  " + fmtMonthLabel_(prevMonth) + ": " + CURRENCY + fmt(prevData.total);
        body += " → " + fmtMonthLabel_(month) + ": " + CURRENCY + fmt(d.total);
        body += " (" + (change >= 0 ? "+" : "-") + changePct + "%)\n";
    }

    var subject = "[RupiBook] 📊 Monthly Summary — " + label;
    MailApp.sendEmail(Session.getEffectiveUser().getEmail(), subject, body);
}

/***
 * ═══════════════════════════════════════════
 *  2. CATEGORY BUDGET ALERT
 *  Subject: [RupiBook] 🚨 Budget Exceeded — Food — Jun 2026
 * ═══════════════════════════════════════════
 ***/
function sendCategoryBudgetAlert(ss, month, category) {
    var d = getMonthDetails_(ss, month);
    var label = fmtMonthLabel_(month);
    var catSpent = d.byCat[category] || 0;
    var catLimit = d.catLimits[category] || 0;

    if(catLimit <= 0) return; // No limit set for this category

    var pct = Math.round(catSpent / catLimit * 100);
    var over = catSpent - catLimit;

    var body = "";
    body += "🚨 Budget Exceeded: " + category + " — " + label + "\n";
    body += "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━\n\n";
    body += catIcon_(category) + " " + category + ": " + CURRENCY + fmt(catSpent) + " / " + CURRENCY + fmt(catLimit) + " (" + pct + "%)\n";
    body += "   Exceeded by: " + CURRENCY + fmt(over) + "\n\n";

    // Top sources in this category
    var catExpenses = d.expenses.filter(function(e) { return e.category === category; });
    var catSources = {};
    var highest = null;

    catExpenses.forEach(function(e) {
        if(e.source) {
            if(!catSources[e.source]) catSources[e.source] = { amount: 0, count: 0 };
            catSources[e.source].amount += e.amount;
            catSources[e.source].count++;
        }
        if(!highest || e.amount > highest.amount) {
            highest = e;
        }
    });

    var srcArr = Object.keys(catSources).map(function(s) {
        return { name: s, amount: catSources[s].amount, count: catSources[s].count };
    }).sort(function(a, b) { return b.amount - a.amount; });

    if(srcArr.length > 0) {
        body += "── Top Sources in " + category + " ──\n";
        srcArr.forEach(function(s, i) {
            body += "  " + (i + 1) + ". " + s.name + "  " + CURRENCY + fmt(s.amount) + " (" + s.count + " txn)\n";
        });
        body += "\n";
    }

    // Highest single expense
    if(highest) {
        var hDate = highest.timestamp instanceof Date
            ? Utilities.formatDate(highest.timestamp, TIMEZONE, "MMM dd")
            : String(highest.timestamp).substring(0, 10);
        body += "── Highest Single Expense ──\n";
        body += "  " + CURRENCY + fmt(highest.amount) + " — " + (highest.source || category) + " — " + hDate + "\n\n";
    }

    // Recent expenses in this category (last 10)
    var recent = catExpenses.sort(function(a, b) {
        return new Date(b.timestamp) - new Date(a.timestamp);
    }).slice(0, 10);

    if(recent.length > 0) {
        body += "── Recent " + category + " Expenses ──\n";
        recent.forEach(function(e) {
            var eDate = e.timestamp instanceof Date
                ? Utilities.formatDate(e.timestamp, TIMEZONE, "MMM dd")
                : String(e.timestamp).substring(0, 10);
            body += "  " + eDate + "  " + CURRENCY + fmt(e.amount) + "  " + (e.source || "—") + "\n";
        });
    }

    var subject = "[RupiBook] 🚨 Budget Exceeded — " + category + " — " + label;
    MailApp.sendEmail(Session.getEffectiveUser().getEmail(), subject, body);
}

        var line = "  " + catIcon_(c) + " " + c + ": " + CURRENCY + fmt(spent);
        if(limit > 0) {
            line += " / " + CURRENCY + fmt(limit) + " (" + Math.round(spent / limit * 100) + "%)";
            if(spent > limit) {
                line += " ⚠️";
                overCats.push({ name: c, over: spent - limit });
            }
        }
        body += line + "\n";
    });
    body += "\n";

    // Categories over budget
    if(overCats.length > 0) {
        body += "── Categories Over Budget ──\n";
        overCats.forEach(function(c) {
            body += "  " + c.name + ": +" + CURRENCY + fmt(c.over) + "\n";
        });
        body += "\n";
    }

    // Remaining days and daily budget
    var now = new Date();
    var endOfMonth = new Date(now.getFullYear(), now.getMonth() + 1, 0);
    var daysLeft = Math.max(0, endOfMonth.getDate() - now.getDate());

    body += "── Action Needed ──\n";
    body += "  Remaining days in month: " + daysLeft + "\n";
    if(daysLeft > 0) {
        // How much over, so no spending recommended
        body += "  You are " + CURRENCY + fmt(over) + " over budget. Consider reducing non-essential spending.\n";
    }

    var subject = "[RupiBook] 🔴 Overall Budget Exceeded — " + label;
    MailApp.sendEmail(Session.getEffectiveUser().getEmail(), subject, body);
}

// Returns a JSON ContentService response
function json(obj) {

        .setMimeType(ContentService.MimeType.JSON);
}

// Indian number formatting: 1234567 → 12,34,567
function fmt(n) {
    n = Math.round(Number(n) || 0);
    var s = String(n);
    var last3 = s.slice(-3);
    var rest = s.slice(0, -3);
    if(rest !== "") last3 = "," + last3;
    return rest.replace(/\B(?=(\d{2})+(?!\d))/g, ",") + last3;
}

// Returns category→source mapping and payment modes for the PWA dropdowns
function getConfigData() {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName("Config");

    var mapping = {};
    var categories = [];

    if(sheet) {
        var data = sheet.getDataRange().getValues();
        for(var i = 1; i < data.length; i++) {
            var category = String(data[i][0] || "").trim();
            var source = String(data[i][1] || "").trim();

            if(!category || !source) continue;

            if(!mapping[category]) {
                mapping[category] = [];
                categories.push(category);
            }

            mapping[category].push(source);
        }
    }

    var pmSheet = ss.getSheetByName("PaymentModes");
    var paymentModes = [];

    if(pmSheet) {
        var pmData = pmSheet.getDataRange().getValues();
        for(var j = 1; j < pmData.length; j++) {
            var mode = String(pmData[j][0] || "").trim();
            if(mode) paymentModes.push(mode);
        }
    }

    return {
        categories: categories,
        mapping: mapping,
        paymentModes: paymentModes
    };
}

// Returns current month's summary for the dashboard
function getDashboardData() {
    var ss = SpreadsheetApp.getActiveSpreadsheet();

    var expenses = ss.getSheetByName("Expenses");
    var settings = ss.getSheetByName("Settings");

    var month = Utilities.formatDate(
        new Date(),
        TIMEZONE,
        "yyyy-MM"
    );

    var total = 0;
    var categoryTotals = {};

    var rows = expenses.getDataRange().getValues();

    for(var i = 1; i < rows.length; i++) {

        var rowMonth = rows[i][7];

        if(rowMonth instanceof Date) {
            rowMonth = Utilities.formatDate(
                rowMonth,
                TIMEZONE,
                "yyyy-MM"
            );
        }

        if(String(rowMonth) !== month) continue;

        var amount = Number(rows[i][1]) || 0;
        var category = rows[i][2] || "Others";

        total += amount;

        categoryTotals[category] =
            (categoryTotals[category] || 0) + amount;
    }

    var limit =
        Number(settings.getRange("B1").getValue()) || 0;

    // Read category budgets for the dashboard overshoot chart
    var categoryBudgets = {};
    var budgets = ss.getSheetByName("Budgets");
    if(budgets) {
        var bData = budgets.getDataRange().getValues();
        for(var j = 1; j < bData.length; j++) {
            var cn = String(bData[j][0] || "").trim();
            if(cn) categoryBudgets[cn] = Number(bData[j][1]) || 0;
        }
    }

    return {
        month: month,
        totalSpent: total,
        monthlyLimit: limit,
        remaining: limit - total,
        budgetPercent: limit > 0 ?
            Math.round((total / limit) * 100) : 0,
        categoryTotals: categoryTotals,
        categoryBudgets: categoryBudgets
    };
}

// Returns distinct categories, sources, and tags for filter chips
function getFilterValues() {

    var sheet =
        SpreadsheetApp.getActiveSpreadsheet()
        .getSheetByName("Expenses");

    var rows = sheet.getDataRange().getValues();

    var categories = {};
    var sources = {};
    var tags = {};

    for(var i = 1; i < rows.length; i++) {

        var category = String(rows[i][2] || "").trim();
        var source = String(rows[i][3] || "").trim();
        var tagString = String(rows[i][5] || "").trim();

        if(category) categories[category] = true;
        if(source) sources[source] = true;

        if(tagString) {

            tagString
                .split(/\s+/)
                .forEach(function(tag) {

                    tag = tag.trim();

                    if(!tag) return;

                    tags[tag] = true;
                });
        }
    }

    return {
        categories: Object.keys(categories).sort(),
        sources: Object.keys(sources).sort(),
        tags: Object.keys(tags).sort()
    };
}

/***
 * ───────── SEARCH ─────────
 * Server-side filtered search. Accepts query parameters:
 *   q         — free-text search (matches category, source, tags, comments)
 *   category  — exact category match
 *   source    — exact source match
 *   tag       — matches if the tag appears in the tags field
 *   from      — date string (yyyy-MM-dd), inclusive lower bound
 *   to        — date string (yyyy-MM-dd), inclusive upper bound
 *   limit     — max results to return (default 50)
 *
 * Returns the most recent matching expenses (newest first).
 ***/
function searchExpenses(params) {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Expenses");
    var rows = sheet.getDataRange().getValues();

    var q = String(params.q || "").trim().toLowerCase();
    var category = String(params.category || "").trim();
    var source = String(params.source || "").trim();
    var tag = String(params.tag || "").trim().toLowerCase();
    var fromStr = String(params.from || "").trim();
    var toStr = String(params.to || "").trim();
    var limit = Number(params.limit) || 50;

    var fromDate = fromStr ? new Date(fromStr) : null;
    var toDate = toStr ? new Date(toStr + "T23:59:59") : null;

    var results = [];

    // Iterate newest-first
    for(var i = rows.length - 1; i >= 1 && results.length < limit; i--) {
        var rowTs = rows[i][0];
        var rowAmount = Number(rows[i][1]) || 0;
        var rowCategory = String(rows[i][2] || "").trim();
        var rowSource = String(rows[i][3] || "").trim();
        var rowTags = String(rows[i][5] || "").trim();
        var rowComments = String(rows[i][6] || "").trim();

        // Skip blank rows
        if(!rowTs || rowAmount === "") continue;

        // Category filter
        if(category && rowCategory !== category) continue;

        // Source filter
        if(source && rowSource !== source) continue;

        // Tag filter
        if(tag) {
            var tagsLower = rowTags.toLowerCase();
            if(tagsLower.indexOf(tag) === -1) continue;
        }

        // Date range filter
        var rowDate = (rowTs instanceof Date) ? rowTs : new Date(rowTs);
        if(fromDate && rowDate < fromDate) continue;
        if(toDate && rowDate > toDate) continue;

        // Free-text search (matches category, source, tags, comments)
        if(q) {
            var haystack = (rowCategory + " " + rowSource + " " + rowTags + " " + rowComments).toLowerCase();
            if(haystack.indexOf(q) === -1) continue;
        }

        results.push({
            timestamp: rows[i][0],
            amount: rows[i][1],
            category: rowCategory,
            source: rowSource,
            paymentMode: String(rows[i][4] || "").trim(),
            tags: rowTags,
            comments: rowComments
        });
    }

    return results;
}
